## Usage

```jsx
/* Editor */
import { RicosEditor } from 'ricos-editor';
import { pluginDivider } from 'wix-rich-content-plugin-divider';
import 'ricos-editor/dist/styles.min.css';
import 'wix-rich-content-plugin-divider/dist/styles.min.css';
<RicosEditor plugins={[pluginDivider()]} />;

/* Viewer */
import { RicosViewer } from 'ricos-viewer';
import { pluginDivider } from 'wix-rich-content-plugin-divider/dist/module.viewer';
import 'wix-rich-content-plugin-divider/dist/styles.min.css';
<RicosViewer plugins={[pluginDivider()]} />;
```

## Divider Config API

### `type`

controls Divider type

```jsx
{
  type: 'single' | 'double' | 'dashed' | 'dotted',
}
```

Default value:

```jsx
{
  type: 'single',
}
```

### `size`

controls Divider size

```jsx
{
  size: 'small' | 'medium' | 'large',
}
```

Default value:

```jsx
{
  size: 'large',
}
```

### `alignment`

controls Divider alignment

```jsx
{
  alignment: 'left' | 'center' | 'right',
}
```

Default value:

```jsx
{
  alignment: 'center',
}
```
